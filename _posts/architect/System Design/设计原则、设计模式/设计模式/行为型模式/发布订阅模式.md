>发布订阅模式（Publish-Subscribe Pattern）

`发布订阅模式`是一种**行为型设计模式**，它定义了一种**一对多**的**消息传递**机制。
`发布者（Publisher）`发送消息时，*不需要知道有哪些订阅者（Subscriber）*；
`订阅者`接收消息时，*也不需要知道消息来自哪个发布者*。
*两者通过`事件总线（Event Bus）` 进行**解耦通信***。

## 核心概念对比

### 观察者模式 vs 发布订阅模式

| 特性       | 观察者模式           | 发布订阅模式              |
| -------- | --------------- | ------------------- |
| **耦合度**  | 观察者和被观察者*直接关联*  | 发布者和订阅者*完全解耦*       |
| **通信方式** | 直接调用 `notify()` | 通过事件中心（`Event Bus`） |
| **关系**   | 一对一或一对多         | 一对多或多对多             |
| **场景**   | 进程内对象通信         | 跨模块、跨组件通信           |

## TypeScript 实现

### 基础版本

```typescript
// 定义事件回调函数的类型
type EventCallback = (...args: any[]) => void;

// 事件总线类
class EventBus {
  // 存储事件和对应的回调函数列表
  private events: Map<string, EventCallback[]>;

  constructor() {
    this.events = new Map();
  }

  /**
   * 订阅事件
   * @param event 事件名称
   * @param callback 回调函数
   */
  subscribe(event: string, callback: EventCallback): void {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    
    const callbacks = this.events.get(event)!;
    callbacks.push(callback);
  }

  /**
   * 取消订阅
   * @param event 事件名称
   * @param callback 要移除的回调函数
   */
  unsubscribe(event: string, callback: EventCallback): void {
    if (!this.events.has(event)) return;
    
    const callbacks = this.events.get(event)!;
    const index = callbacks.indexOf(callback);
    
    if (index !== -1) {
      callbacks.splice(index, 1);
    }
    
    // 如果没有订阅者了，删除该事件
    if (callbacks.length === 0) {
      this.events.delete(event);
    }
  }

  /**
   * 发布事件
   * @param event 事件名称
   * @param args 传递给回调函数的参数
   */
  publish(event: string, ...args: any[]): void {
    if (!this.events.has(event)) return;
    
    const callbacks = this.events.get(event)!;
    
    // 遍历所有订阅者，执行回调
    callbacks.forEach(callback => {
      try {
        callback(...args);
      } catch (error) {
        console.error(`Error in event handler for "${event}":`, error);
      }
    });
  }

  /**
   * 清空所有事件
   */
  clear(): void {
    this.events.clear();
  }

  /**
   * 获取某个事件的订阅者数量
   */
  getSubscriberCount(event: string): number {
    return this.events.get(event)?.length || 0;
  }
}

// 创建全局事件总线实例（单例模式）
export const eventBus = new EventBus();
```

### 增强版本（带类型安全）

```typescript
// 定义事件映射接口，提供类型安全
interface EventMap {
  'user:login': (userId: string, userInfo: UserInfo) => void;
  'user:logout': (userId: string) => void;
  'order:created': (orderId: string, amount: number) => void;
  'order:updated': (orderId: string, status: string) => void;
  'notification:message': (message: string, type: 'info' | 'warning' | 'error') => void;
}

interface UserInfo {
  id: string;
  name: string;
  email: string;
}

// 类型安全的事件总线
class TypedEventBus {
  private events: Map<keyof EventMap, Set<Function>>;

  constructor() {
    this.events = new Map();
  }

  subscribe<K extends keyof EventMap>(event: K, callback: EventMap[K]): () => void {
    if (!this.events.has(event)) {
      this.events.set(event, new Set());
    }

    this.events.get(event)!.add(callback as Function);

    // 返回取消订阅的函数
    return () => {
      this.unsubscribe(event, callback);
    };
  }

  unsubscribe<K extends keyof EventMap>(event: K, callback: EventMap[K]): void {
    const callbacks = this.events.get(event);
    if (callbacks) {
      callbacks.delete(callback as Function);
      
      if (callbacks.size === 0) {
        this.events.delete(event);
      }
    }
  }

  publish<K extends keyof EventMap>(event: K, ...args: Parameters<EventMap[K]>): void {
    const callbacks = this.events.get(event);
    
    if (callbacks) {
      callbacks.forEach(callback => {
        try {
          (callback as EventMap[K])(...args);
        } catch (error) {
          console.error(`Error in handler for "${event}":`, error);
        }
      });
    }
  }

  // 只触发一次
  once<K extends keyof EventMap>(event: K, callback: EventMap[K]): void {
    const wrapper: EventMap[K] = ((...args: Parameters<EventMap[K]>) => {
      callback(...args);
      this.unsubscribe(event, wrapper);
    }) as EventMap[K];
    
    this.subscribe(event, wrapper);
  }
}

export const typedEventBus = new TypedEventBus();
```

```typescript
// 定义事件
interface AuthEvents {
  'auth:login': (user: User) => void;
  'auth:logout': () => void;
  'auth:tokenRefresh': (newToken: string) => void;
}

interface User {
  id: string;
  name: string;
  role: string;
}

class AuthService {
  private eventBus: TypedEventBus;

  constructor(eventBus: TypedEventBus) {
    this.eventBus = eventBus;
  }

  login(username: string, password: string): void {
    // 模拟登录逻辑
    const user: User = { id: '1', name: username, role: 'admin' };
    
    console.log(`User ${username} logged in`);
    
    // 发布登录成功事件
    this.eventBus.publish('auth:login', user);
  }

  logout(): void {
    console.log('User logged out');
    this.eventBus.publish('auth:logout');
  }
}

class LoggerService {
  constructor(eventBus: TypedEventBus) {
    eventBus.subscribe('auth:login', (user) => {
      console.log(`[LOG] User logged in: ${user.name}`);
    });

    eventBus.subscribe('auth:logout', () => {
      console.log('[LOG] User logged out');
    });
  }
}

  constructor(eventBus: TypedEventBus) {

    });
  }
}
```

## 高级特性实现

### 1. 支持优先级

```typescript
interface PriorityCallback {
  callback: EventCallback;
  priority: number; // 数字越大，优先级越高
}

class PriorityEventBus {
  private events: Map<string, PriorityCallback[]>;

  constructor() {
    this.events = new Map();
  }

  subscribe(event: string, callback: EventCallback, priority: number = 0): void {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }

    const callbacks = this.events.get(event)!;
    callbacks.push({ callback, priority });
    
    // 按优先级排序
    callbacks.sort((a, b) => b.priority - a.priority);
  }

  publish(event: string, ...args: any[]): void {
    const callbacks = this.events.get(event);
    if (!callbacks) return;

    callbacks.forEach(({ callback }) => {
      callback(...args);
    });
  }
}

// 使用示例
const priorityBus = new PriorityEventBus();

priorityBus.subscribe('data:loaded', () => console.log('Low priority handler'), 1);
priorityBus.subscribe('loaded', () => console.log('High priority handler'), 10);
priorityBus.subscribe('data:loaded', () => console.log('Medium priority handler'), 5);

priorityBus.publish('data:loaded');
// 输出顺序:
// High priority handler
// Medium priority handler
// Low priority handler
```

### 2. 支持异步处理

```typescript
type AsyncEventCallback = (...args: any[]) => Promise<void> | void;

class AsyncEventBus {
  private events: Map<string, AsyncEventCallback[]>;

  constructor() {
    this.events = new Map();
  }

  subscribe(event: string, callback: AsyncEventCallback): void {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event)!.push(callback);
  }

  async publishAsync(event: string, ...args: any[]): Promise<void> {
    const callbacks = this.events.get(event);
    if (!callbacks) return;

    // 并行执行所有异步处理器
    await Promise.all(
      callbacks.map(callback => {
        try {
          return callback(...args);
        } catch (error) {
          console.error(`Error in async handler for "${event}":`, error);
          return Promise.resolve();
        }
      })
    );
  }

  // 串行执行
  async publishSeries(event: string, ...args: any[]): Promise<void> {
    const callbacks = this.events.get(event);
    if (!callbacks) return;

    for (const callback of callbacks) {
      try {
        await callback(...args);
      } catch (error) {
        console.error(`Error in series handler for "${event}":`, error);
      }
    }
  }
}
```

## 优缺点分析

### ✅ 优点

1. **解耦**：*发布者和订阅者完全独立，互不知道对方存在*
2. **灵活性**：可以动态添加/移除订阅者
3. **可扩展性**：新增功能只需添加新的订阅者，无需修改发布者
4. **支持广播**：一次发布，多个订阅者接收
5. **支持异步**：可以异步处理消息，提高性能

### ❌ 缺点

1. **内存泄漏风险**：订阅者忘记取消订阅会导致内存泄漏
2. **调试困难**：消息流向不直观，难以追踪
3. **性能问题**：订阅者过多时，发布消息可能变慢
4. **消息处理顺序**：默认不保证处理顺序（除非实现优先级）
5. **错误处理**：一个订阅者报错可能影响其他订阅者

## 最佳实践

```typescript
// 1. 使用 WeakMap 避免内存泄漏
class SafeEventBus {
  private events: Map<string, WeakSet<EventCallback>>;
  
  // 2. 提供取消订阅的方法
  subscribe(event: string, callback: EventCallback): () => void {
    // ... 订阅逻辑
    
    // 返回取消订阅函数
    return () => this.unsubscribe(event, callback);
  }

  // 3. 错误隔离
  publish(event: string, ...args: any[]): void {
    const callbacks = this.events.get(event);
    if (!callbacks) return;

    callbacks.forEach(callback => {
      try {
        callback(...args);
      } catch (error) {
        console.error(`Handler error:`, error);
        // 不影响其他订阅者
      }
    });
  }

  // 4. 提供 once 方法
  once(event: string, callback: EventCallback): void {
    const wrapper = (...args: any[]) => {
      callback(...args);
      this.unsubscribe(event, wrapper);
    };
    this.subscribe(event, wrapper);
  }
}

// 5. 在 React/Vue 组件中正确清理
class MyComponent {
  private unsubscribe: () => void;

  constructor(eventBus: TypedEventBus) {
    this.unsubscribe = eventBus.subscribe('some:event', this.handler);
  }

  // 组件销毁时清理
  destroy(): void {
    this.unsubscribe();
  }

  private handler(data: any): void {
    // 处理逻辑
  }
}
```

## 总结

发布订阅模式是前端开发中**最常用**的设计模式之一，广泛应用于：

- **组件通信**（Vue 的 EventBus、React 的 Context + Hooks）
- **状态管理**（Redux、Vuex 的订阅机制）
- **消息队列**（WebSocket、Server-Sent Events）
- **事件驱动架构**（Node.js 的 EventEmitter）

掌握这个模式，能让你写出**更解耦、更易维护**的代码！
