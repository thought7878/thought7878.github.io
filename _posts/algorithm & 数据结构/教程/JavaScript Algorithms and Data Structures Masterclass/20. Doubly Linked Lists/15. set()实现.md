  - index边界情况：index<0||index>=length return null
  - 获取index节点node：get(index)

  */
  set(index, val) {
    if (index < 0 || index >= this.length) return null;

    let node = this.get(index);
    if (node) {
      node.value = val;
      return node;
    }
    return null;
  }
