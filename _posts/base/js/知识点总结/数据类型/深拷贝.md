手写一个健壮的深拷贝函数是*前端面试的“终极试炼”之一*。一个合格的深拷贝不仅要处理普通的 Object 和 Array，还要解决**循环引用、Symbol 键、特殊内置对象（Date/RegExp/Map/Set）、原型链丢失**等边界问题。

以下是一份**面试满分、生产级别**的深拷贝实现，附带逐行详细注释。

---

### 🏆 完整实现代码

```javascript
/**
 * 辅助函数：判断是否为引用类型（对象、函数、数组）
 */
function isObject(target) {
  const type = typeof target;
  return target !== null && (type === 'object' || type === 'function');
}

/**
 * 辅助函数：获取数据的精确类型
 */
function getType(target) {
  return Object.prototype.toString.call(target).slice(8, -1);
}

/**
 * 核心深拷贝函数
 * @param {any} target - 需要深拷贝的目标数据
 * @param {WeakMap} map - 用于记录已拷贝的对象，解决循环引用问题
 * @returns {any} - 拷贝后的新数据
 */
function deepClone(target, map = new WeakMap()) {
  // 1. 基本类型（String, Number, Boolean, Null, Undefined, Symbol, BigInt）直接返回原值
  if (!isObject(target)) {
    return target;
  }

  // 2. 获取精确类型，用于后续分类处理
  const type = getType(target);

  // 3. 处理特殊的内置对象（这些对象不需要递归遍历其内部属性）
  if (type === 'Date') return new Date(target); // 拷贝日期对象
  if (type === 'RegExp') return new RegExp(target.source, target.flags); // 拷贝正则对象（保留修饰符）
  if (type === 'Error') {
    // 拷贝错误对象，保留 message 和 stack
    const err = new target.constructor(target.message);
    err.stack = target.stack; 
    return err;
  }
  
  // 函数通常不需要深拷贝，直接返回原引用（若需拷贝函数体可使用 new Function，但不推荐）
  if (type === 'Function') return target; 
  
  // WeakMap 和 WeakSet 的键是弱引用且不可遍历，无法深拷贝，直接返回原引用
  if (type === 'WeakMap' || type === 'WeakSet') return target;

  // 4. 解决循环引用问题（核心考点）
  // 如果 map 中已经存在该对象，说明遇到了循环引用，直接返回之前拷贝的实例，防止栈溢出
  if (map.has(target)) {
    return map.get(target);
  }

  // 5. 初始化目标容器，并严格保持原有的原型链
  let cloneTarget;
  if (type === 'Array') {
    cloneTarget = [];
  } else if (type === 'Map') {
    cloneTarget = new Map();
  } else if (type === 'Set') {
    cloneTarget = new Set();
  } else {
    // 对于普通对象或自定义类的实例，使用 Object.create 保持其原型链（__proto__）
    // 这样拷贝出来的对象依然能 instanceof 原来的构造函数
    cloneTarget = Object.create(Object.getPrototypeOf(target));
  }

  // 6. 将当前对象记录到 map 中（必须在递归遍历属性之前记录，否则无法拦截死循环）
  map.set(target, cloneTarget);

  // 7. 针对 Map 和 Set 进行特殊的递归拷贝
  if (type === 'Map') {
    target.forEach((value, key) => {
      // Map 的 key 和 value 都可能是引用类型，都需要深拷贝
      cloneTarget.set(deepClone(key, map), deepClone(value, map));
    });
    return cloneTarget;
  }

  if (type === 'Set') {
    target.forEach(value => {
      cloneTarget.add(deepClone(value, map));
    });
    return cloneTarget;
  }

  // 8. 处理 Array 和 Object（包含 Symbol 键和不可枚举键）
  // 使用 Reflect.ownKeys 替代 Object.keys，因为它能同时获取字符串键和 Symbol 键
  Reflect.ownKeys(target).forEach(key => {
    // 获取原对象的属性描述符（包含 writable, enumerable, configurable 等配置）
    const desc = Object.getOwnPropertyDescriptor(target, key);
    
    if (desc) {
      if ('value' in desc) {
        // 普通数据属性：递归拷贝值
        cloneTarget[key] = deepClone(target[key], map);
      } else {
        // 访问器属性（getter/setter）：
        // 深拷贝的语义通常是“拷贝当前的值”，而不是拷贝 getter 函数本身。
        // 因此我们将 target[key]（触发 getter 获取当前值）作为新值，并保留其他描述符配置。
        Object.defineProperty(cloneTarget, key, {
          ...desc,
          value: deepClone(target[key], map) 
        });
      }
    }
  });

  return cloneTarget;
}
```

### 🔍 核心难点与亮点解析（面试加分项）

#### 1. 循环引用的处理（`WeakMap`）
* **痛点**：如果对象 `a` 的属性指向 `b`，`b` 的属性又指向 `a`，普通的递归会导致无限死循环，最终 `Maximum call stack size exceeded`。
* **解法**：使用 `WeakMap` 作为哈希表，`key` 存原对象，`value` 存拷贝后的新对象。每次递归前先检查 `map.has(target)`。
* **为什么用 `WeakMap` 而不是 `Map`？** `WeakMap` 的键是弱引用，不会阻止垃圾回收。当原对象被销毁时，`WeakMap` 中的记录会自动清除，**防止内存泄漏**。

#### 2. 原型链的保持（`Object.create`）
* **痛点**：使用 `{}` 或 `new Object()` 创建新对象，会导致自定义类的实例丢失原型链（`instanceof` 失效）。
* **解法**：使用 `Object.create(Object.getPrototypeOf(target))`，直接以原对象的原型作为新对象的原型，完美复刻继承关系。

#### 3. Symbol 键与不可枚举属性（`Reflect.ownKeys`）
* **痛点**：`for...in` 和 `Object.keys()` 只能获取可枚举的字符串键，会漏掉 `Symbol` 键和不可枚举属性。
* **解法**：使用 `Reflect.ownKeys(target)`，它等价于 `Object.getOwnPropertyNames(target)` + `Object.getOwnPropertySymbols(target)`，能获取对象**自身的所有属性键**。

#### 4. 属性描述符的保留（`Object.getOwnPropertyDescriptor`）
* **痛点**：直接 `clone[key] = deepClone(target[key])` 会丢失原属性的 `writable`（是否可写）、`enumerable`（是否可枚举）等配置。
* **解法**：获取原描述符，并使用 `Object.defineProperty` 重新定义，确保拷贝后的对象在行为上与原对象完全一致。

### 🧪 极限测试用例

```javascript
// 1. 基础类型与 Symbol 键
const sym = Symbol('key');
const obj1 = { a: 1, [sym]: 'symbol value' };

// 2. 特殊内置对象
const obj2 = { 
  date: new Date(), 
  reg: /abc/gi, 
  err: new Error('test error') 
};

// 3. Map 和 Set
const map = new Map();
map.set('key', { nested: true });
const set = new Set([1, 2, { nested: true }]);
const obj3 = { map, set };

// 4. 循环引用 (终极考验)
const obj4 = { name: 'circular' };
obj4.self = obj4; 

// 5. 自定义类与原型链
class Person {
  constructor(name) { this.name = name; }
  sayHi() { console.log(`Hi, ${this.name}`); }
}
const obj5 = new Person('Alice');

// 执行深拷贝
const clone1 = deepClone(obj1);
const clone2 = deepClone(obj2);
const clone3 = deepClone(obj3);
const clone4 = deepClone(obj4);
const clone5 = deepClone(obj5);

// 验证结果
console.log(clone1[sym]); // 'symbol value' (Symbol 键被正确拷贝)
console.log(clone2.date instanceof Date); // true
console.log(clone3.map.get('key') !== obj3.map.get('key')); // true (Map 内部对象被深拷贝)
console.log(clone4.self === clone4); // true (循环引用被正确解开并指向新对象)
console.log(clone5 instanceof Person); // true (原型链被完美保留)
clone5.sayHi(); // 'Hi, Alice'
```

### 💡 现代 JS 的原生替代方案：`structuredClone`

在实际工程开发中，**强烈建议不要自己手写深拷贝**，而是使用 HTML5 规范引入、现代浏览器和 Node.js (v17+) 已原生支持的 `structuredClone()` 函数。

```javascript
const original = { a: 1, b: { c: 2 } };
original.self = original; // 循环引用

// 原生深拷贝，自带循环引用处理、支持 Map/Set/Date/RegExp 等
const cloned = structuredClone(original); 
```

**`structuredClone` 的优势与限制：**
* **优势**：C++ 底层实现，性能极高；原生支持循环引用和多种复杂数据类型（如 `Blob`, `File`, `ArrayBuffer`）。
* **限制**：**不能拷贝函数（Function）**，不能拷贝 DOM 节点，不能拷贝原型链（拷贝后的对象原型会丢失，变成纯 Object）。

**总结**：
* **面试时**：手写上述代码，展示你对 JS 底层机制（原型、描述符、WeakMap、Reflect）的深刻理解。
* **工作中**：优先使用 `structuredClone`；如果需要拷贝函数或保留原型链，再考虑使用 `lodash.cloneDeep` 或上述手写方案。
