# 京东商智数据分析平台 - 详细设计文档

## 1. 文档概述

### 1.1 文档目的

本文档是京东商智数据分析平台的详细设计文档，基于概要设计文档细化每个模块的类级别设计、接口定义、数据模型、核心流程、异常处理和测试策略，为开发编码提供直接指导。

### 1.2 适用范围

- 后端：NestJS + TypeScript + Prisma
- 前端：React 18+ + TypeScript + Ant Design
- 数据库：PostgreSQL 15+
- 缓存：Redis 7+

### 1.3 设计原则

- 每个模块独立设计，可独立测试
- 接口明确定义输入输出和异常
- 前端组件与后端服务解耦
- 测试策略覆盖核心业务逻辑

## 2. 全局约定

### 2.1 统一响应格式

所有 API 接口返回统一格式：

```json
{
  "code": 200,
  "message": "success",
  "data": {},
  "timestamp": 1690000000000
}
```

错误响应：

```json
{
  "code": 4001,
  "message": "参数校验失败：email 格式不正确",
  "data": null,
  "timestamp": 1690000000000
}
```

### 2.4 多租户中间件约定

所有请求通过 `TenantMiddleware` 自动注入 `tenantId`：

```typescript
// 从 JWT payload 中提取 tenantId，注入到 request 对象
@Injectable()
export class TenantMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    const user = req.user; // from JWT
    (req as any).tenantId = user.tenantId;
    next();
  }
}
```

所有业务查询自动追加 `WHERE tenant_id = ?`。

#### 5.1.6 核心流程图

```
登录流程:
1. 接收 LoginDto (account + password)
2. 查询用户 (username OR email OR phone)
   ├─ 用户不存在 → 抛出错误 (1001: 账号或密码错误)
   └─ 用户存在
       ├─ 验证密码
       │   ├─ 密码错误 → 抛出错误 (1001: 账号或密码错误)
       │   └─ 密码正确
       │       ├─ 检查用户状态
       │       │   ├─ disabled → 抛出错误 (1002: 账号已禁用)
       │       │   └─ active → 继续
       │       ├─ 生成 AccessToken (2h 过期)
       │       ├─ 生成 RefreshToken (7d 过期)
       │       ├─ 更新 lastLoginAt, lastLoginIp
       │       ├─ 记录登录日志
       │       └─ 返回 LoginResponse
```

// src/store/authStore.ts
interface AuthState {
  isAuthenticated: boolean;
  user: UserInfo | null;
  accessToken: string | null;
  refreshToken: string | null;

  // Actions
  login: (credentials: LoginRequest) => Promise<void>;
  logout: () => Promise<void>;
  refreshToken: () => Promise<void>;
  setUser: (user: UserInfo) => void;
  clearAuth: () => void;
}

export const useAuthStore = create<AuthState>((set, get) => ({
  isAuthenticated: false,
  user: null,
  accessToken: null,
  refreshToken: null,

  login: async (credentials) => {
    const res = await authApi.login(credentials);
    set({
      isAuthenticated: true,
      user: res.user,
      accessToken: res.accessToken,
      refreshToken: res.refreshToken,
    });
    persistAuth(res);
  },

  logout: async () => {
    await authApi.logout();
    get().clearAuth();
  },

  refreshToken: async () => {
    const rt = get().refreshToken;
    if (!rt) throw new Error('No refresh token');
    const res = await authApi.refreshToken(rt);
    set({ accessToken: res.accessToken });
    persistAuth({ accessToken: res.accessToken });
  },

  setUser: (user) => set({ user }),
  clearAuth: () => {
    set({
      isAuthenticated: false,
      user: null,
      accessToken: null,
      refreshToken: null,
    });
    clearPersistedAuth();
  },
}));

      }

// src/hooks/useDebounce.ts
export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);
  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);
  return debouncedValue;
}
